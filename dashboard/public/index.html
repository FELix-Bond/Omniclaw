ONE-TIME SETUP
git clone --recursive https://github.com/microsoft/BitNet
cd BitNet && pip install -r requirements.txt
python setup_env.py --hf-repo microsoft/BitNet-b1.58-2B-4T --quant-type i2_s
# then run inference:
python run_inference.py -m models/BitNet-b1.58-2B-4T/ggml-model-i2_s.gguf -p "your prompt"
